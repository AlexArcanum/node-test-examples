Get the schema

{User} = require "./models"
{validateAge} = require "./helpers"

getUser = (data, callback) ->
  if not data.name 
    return callback "Name required"

  query = User.findOne {}

  query.where "name", data.name
  
  if data.age
    query.where "age", data.age

  query.exec callback
